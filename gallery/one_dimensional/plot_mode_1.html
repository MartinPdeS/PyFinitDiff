1D Finit-difference parameters¶

boundaries: {left: symmetric, right: symmetric}

16 from scipy.sparse import linalg
17
18 from PyFinitDiff.Sparse1D import FiniteDifference1D
19 from PyFinitDiff.Utils import get_1D_circular_mesh_triplet
20 from MPSPlots.Render2D import Scene2D, Axis, Line
21
22
23 n_x = 200
24 sparse_instance = FiniteDifference1D(n_x=n_x,
25                                      dx=1,
26                                      derivative=2,
27                                      accuracy=6,
28                                      boundaries={'left': 'zero',
29                                                  'right': 'symmetric'})
30
31 mesh_triplet = get_1D_circular_mesh_triplet(n_x=n_x, radius=60, value0=1, value1=1.4444, x_offset=-100)
32
33 dynamic_triplet = sparse_instance.triplet + mesh_triplet
34
35 eigen_values, eigen_vectors = linalg.eigs(dynamic_triplet.to_dense(), k=4, which='LM', sigma=1.4444)
36
37 figure = Scene2D(unit_size=(3, 3), tight_layout=True)
38
39 for i in range(4):
40     Vector = eigen_vectors[:, i].real.reshape([sparse_instance.n_x])
41     ax = Axis(row=0, col=i, title=f'eigenvalues: \n{eigen_values[i]:.3f}')
42     artist = Line(y=Vector)
43     ax.add_artist(artist)
44     figure.add_axes(ax)
45
46 figure.show()
